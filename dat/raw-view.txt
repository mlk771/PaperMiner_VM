function updateDateRange () {
	
	var start = parseInt($('#rawview-year-start').val());
	var end = parseInt($('#rawview-year-end').val());
	//alert(start.toString() + end.toString());
	if (start >= 1800 && end <= 2000 && start <= end)
	{
//		$('#y2k-slider').val(start.toString() + ';' + end.toString());
		$( "#y2k-slider" ).slider("value", start.toString() + ';' + end.toString());
	//var span = $('#y2k-slider').val().split(';');
		 // alert($('#y2k-slider').val().split(';').toString());
	}
}
